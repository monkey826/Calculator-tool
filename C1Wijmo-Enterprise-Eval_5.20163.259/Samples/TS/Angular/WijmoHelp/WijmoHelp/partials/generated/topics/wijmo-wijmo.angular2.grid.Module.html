Contains Angular 2 components for the wijmo.grid module.

wijmo.angular2.grid is an external TypeScript module that can be imported to your code using its ambient module name. For example:

<p>Here is a data bound FlexGrid control with four columns:</p>
<wj-flex-grid [itemsSource]="data">
  <wj-flex-grid-column 
    [header]="'Country'" 
    [binding]="'country'">
  </wj-flex-grid-column>
  <wj-flex-grid-column 
    [header]="'Sales'" 
    [binding]="'sales'">
  </wj-flex-grid-column>
  <wj-flex-grid-column 
    [header]="'Expenses'" 
    [binding]="'expenses'">
  </wj-flex-grid-column>
  <wj-flex-grid-column 
    [header]="'Downloads'" 
    [binding]="'downloads'">
  </wj-flex-grid-column>
</wj-flex-grid>
