wijmo/wijmo.angular2.viewer

Contains Angular 2 components for the wijmo.viewer module.

wijmo.angular2.viewer is an external TypeScript module that can be imported to your code using its ambient module name. For example:

import * as wjViewer from 'wijmo/wijmo.angular2.viewer';
 
@Component({
    directives: [wjViewer.WjReportViewer, wjViewer.WjPdfViewer],
    template: `
      <wj-report-viewer [reportName]="sales" [serviceUrl]="'webserviceApi'">
      </wj-report-viewer;`,
    selector: 'my-cmp',
})
export class MyCmp {
    data: any[];
}
